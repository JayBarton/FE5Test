#version 330 core
in vec2 TextureCoordinates;
out vec4 color;

uniform sampler2D image;
uniform vec4 spriteColor;
uniform float subtractValue;

void main()
{    
    color = vec4(spriteColor) * texture(image, TextureCoordinates);
	
	if(subtractValue > 0)
	{
		color -= vec4(vec3(subtractValue/255.0f), 0);
	}
}
